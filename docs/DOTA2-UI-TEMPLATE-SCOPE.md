# Dota2 UI Template Scope

## 1. 文档目的

本文档定义 Rune Weaver 三个核心 UI Pattern 在 Dota2 宿主中的最小模板结构。

目标：
- 让 `ui.selection_modal` / `ui.key_hint` / `ui.resource_bar` 具备可执行的模板边界
- 明确 TSX/LESS 最小结构、数据接口、host bridge 依赖
- 为下一轮 UI Adapter 实现提供清晰依据

## 2. 模板文件组织

### 2.1 目录结构

```
content/panorama/src/rune_weaver/generated/ui/
├── index.tsx                    # UI 索引 (refresh)
├── {featureId}.tsx              # Feature TSX 组件
├── {featureId}.less             # Feature LESS 样式
└── types.d.ts                   # 共享类型定义 (可选)
```

### 2.2 文件职责

| 文件 | 生成方式 | 内容 |
|------|----------|------|
| `index.tsx` | refresh | 聚合所有 active features 的导出 |
| `{featureId}.tsx` | create/update | 单个 feature 的 TSX 组件 |
| `{featureId}.less` | create/update | 单个 feature 的样式 |

## 3. `ui.selection_modal` 模板

### 3.1 TSX 最小结构

```tsx
// {featureId}.tsx
// Generated by Rune Weaver
// pattern: ui.selection_modal
// featureId: {featureId}

import React, { useState, useEffect } from "react";

// === Types ===
interface SelectionOption {
  id: string;
  name: string;
  description: string;
  icon?: string;
}

interface {FeatureName}Props {
  title: string;
  options: SelectionOption[];
  timeRemaining?: number;  // 倒计时秒数，可选
  onSelect: (optionId: string) => void;
}

// === Component ===
export function {FeatureName}(props: {FeatureName}Props): React.ReactElement {
  const { title, options, timeRemaining, onSelect } = props;
  const [selectedId, setSelectedId] = useState<string | null>(null);
  const [remaining, setRemaining] = useState(timeRemaining);

  // 倒计时逻辑
  useEffect(() => {
    if (!timeRemaining || timeRemaining <= 0) return;
    
    const timer = setInterval(() => {
      setRemaining((prev) => {
        if (prev === undefined || prev <= 1) {
          clearInterval(timer);
          return 0;
        }
        return prev - 1;
      });
    }, 1000);

    return () => clearInterval(timer);
  }, [timeRemaining]);

  const handleSelect = (optionId: string) => {
    if (selectedId) return;  // 已选择则禁用
    setSelectedId(optionId);
    onSelect(optionId);
  };

  return (
    <Panel className="rw-selection-modal">
      {/* 标题 */}
      <Label className="rw-selection-title" text={title} />
      
      {/* 倒计时 */}
      {remaining !== undefined && remaining > 0 && (
        <Label className="rw-selection-timer" text={`${remaining}s`} />
      )}
      
      {/* 选项列表 */}
      <Panel className="rw-selection-options">
        {options.map((option) => (
          <Panel
            key={option.id}
            className={`rw-selection-option ${selectedId === option.id ? 'selected' : ''} ${selectedId && selectedId !== option.id ? 'disabled' : ''}`}
            onactivate={() => handleSelect(option.id)}
          >
            {option.icon && (
              <Image className="rw-option-icon" src={option.icon} />
            )}
            <Label className="rw-option-name" text={option.name} />
            <Label className="rw-option-description" text={option.description} />
          </Panel>
        ))}
      </Panel>
    </Panel>
  );
}

// === NetTable Integration ===
// 实际使用时需包装 NetTable 监听层
export function {FeatureName}WithNetTable(): React.ReactElement | null {
  // 从 NetTable 获取数据
  const data = useNetTable<{ options: SelectionOption[]; title: string }>(
    "rw_selection_{featureId}",
    "data"
  );
  
  if (!data) return null;
  
  const handleSelect = (optionId: string) => {
    GameEvents.SendCustomGameEventToServer("{featureId}_select", {
      optionId,
    });
  };
  
  return (
    <{FeatureName}
      title={data.title}
      options={data.options}
      onSelect={handleSelect}
    />
  );
}
```

## 4. `ui.key_hint` 模板

### 4.1 TSX 最小结构

```tsx
// {featureId}.tsx
// Generated by Rune Weaver
// pattern: ui.key_hint

import React from "react";

// === Types ===
interface {FeatureName}Props {
  keyBinding: string;      // 按键，如 "Q", "W", "F"
  abilityName: string;     // 技能名称
  visible?: boolean;       // 是否显示
  cooldown?: number;       // 冷却剩余秒数
}

// === Component ===
export function {FeatureName}(props: {FeatureName}Props): React.ReactElement | null {
  const { keyBinding, abilityName, visible = true, cooldown } = props;
  
  if (!visible) return null;
  
  return (
    <Panel className="rw-key-hint">
      <Panel className={`rw-key-badge ${cooldown && cooldown > 0 ? 'on-cooldown' : ''}`}>
        <Label className="rw-key-text" text={keyBinding} />
        {cooldown && cooldown > 0 && (
          <Label className="rw-cooldown-text" text={Math.ceil(cooldown).toString()} />
        )}
      </Panel>
      <Label className="rw-ability-name" text={abilityName} />
    </Panel>
  );
}

// === Position Variants ===
// 不同位置的包装组件
export function {FeatureName}BottomRight(props: {FeatureName}Props): React.ReactElement | null {
  return (
    <Panel className="rw-key-hint-container bottom-right">
      <{FeatureName} {...props} />
    </Panel>
  );
}
```

## 5. `ui.resource_bar` 模板

### 5.1 TSX 最小结构

```tsx
// {featureId}.tsx
// Generated by Rune Weaver
// pattern: ui.resource_bar

import React from "react";

// === Types ===
interface {FeatureName}Props {
  current: number;
  max: number;
  label?: string;          // 资源名称，如 "能量"
  color?: string;          // 主色调，如 "#4a90d9"
  showValue?: boolean;     // 是否显示数值
}

// === Component ===
export function {FeatureName}(props: {FeatureName}Props): React.ReactElement {
  const { 
    current, 
    max, 
    label = "Resource", 
    color = "#4a90d9",
    showValue = true 
  } = props;
  
  const percentage = max > 0 ? Math.min((current / max) * 100, 100) : 0;
  
  return (
    <Panel className="rw-resource-bar">
      {label && (
        <Label className="rw-resource-label" text={label} />
      )}
      
      <Panel className="rw-resource-track">
        <Panel 
          className="rw-resource-fill"
          style={{
            backgroundColor: color,
            width: `${percentage}%`,
          }}
        />
        
        {/* 低资源警告效果 */}
        {percentage < 20 && (
          <Panel className="rw-resource-warning" />
        )}
      </Panel>
      
      {showValue && (
        <Label 
          className="rw-resource-value" 
          text={`${Math.floor(current)}/${max}`} 
        />
      )}
    </Panel>
  );
}
```
